Hash Table

LeetCode

If the key's range is not too large, array is a perfect hash-table.

If the key's range is small enough, a char/int integer is a perfect hash-table.

A simple hash table implemented with C (if not mentioned, solutions are based on the following definitions.):

struct He {
    int val;
    struct He *next;
};
struct Ht {
    int size;
    struct He **list;
};
struct Ht *createHT(void) {
    struct Ht *ht = malloc(sizeof(struct Ht));
    ht->size = 1000;
    ht->list = calloc(ht->size, sizeof(struct He *));
    return ht;
}
void freeHt(struct Ht *ht) {
    struct He *p, *q;
    int i;
    for (i = 0; i < ht->size; i++) {
        p = ht->list[i];
        while (p) {
            q = p->next;
            free(p);
            p = q;
        }
    }
    free(ht->list);
    free(ht);
}
void deleteHT(struct Ht *ht, int val) {
    int ind = val % ht->size;
    if (ind < 0) ind = -ind;
    struct He **p = &(ht->list[ind]);
    while (*p) {
        if ((*p)->val == val) {
            struct He *tmp = (*p)->next;
            free(*p);
            *p = tmp;
            return;
        }
        *p = (*p)->next;
    }
}
bool find_or_insert(struct Ht *ht, int val) {
    int ind = val % ht->size;
    if (ind < 0) ind = -ind;
    struct He *p = ht->list[ind];
    while (p) {
        if (p->val == val) return true;
        p = p->next;
    }
    p = malloc(sizeof(struct He));
    p->val = val;
    p->next = ht->list[ind];
    ht->list[ind] = p;
    return false;
}

217. Contains Duplicate

Solution:

bool containsDuplicate(int* nums, int numsSize) {
    struct Ht *ht = createHT();
    int i;
    for (i = 0; i < numsSize; i++) {
        if (find_or_insert(ht, nums[i])) return true;
    }
    return false;
}

219. Contains Duplicate ||

Solution:

bool containsNearbyDuplicate(int* nums, int numsSize, int k) {
    if (k < 1) return false;
    struct Ht *ht = createHT();
    int i;
    for (i = 0; i < numsSize; i++) {
        if (find_or_insert(ht, nums[i])) return true;
        if (i >= k) {
            deleteHT(ht, nums[i - k]);
        }
    }
    return false;
}

389. Find the Difference

Common solution:

char findTheDifference(char* s, char* t) {
    int ht[26] = {0};
    char *p;
    for (p = s; *p; p++) ht[*p - 'a']++;
    for (p = t; *p; p++) ht[*p - 'a']--;
    int i;
    for (i = 0; i < 26 && !ht[i]; i++);
    return i + 'a';
}

Using a 32 bits unsigned integer instead of the array.

char findTheDifference(char* s, char* t) {
    uint32_t ht = 0, flag = 1;
    char *p;
    for (p = s; *p; p++) ht ^= flag << (*p - 'a');
    for (p = t; *p; p++) ht ^= flag << (*p - 'a');
    char c = 'a';
    while (ht >>= 1) c++;
    return c;
}

Bit manipulation (but not a technique of hash table):

char findTheDifference(char* s, char* t) {
    char *p, c = 0;
    for (p = s; *p; p++) c ^= *p;
    for (p = t; *p; p++) c ^= *p;
    return c;
}

409. Longest Palindrome

Solution:

int longestPalindrome(char* s) {
    int ht[256] = {0}, res = 0;
    char *p;
    for (p = s; *p; p++) {
        if (ht[*p]) {
            res += 2;
            ht[*p] = 0;
        }
        else ht[*p]++;
    }
    int i; 
    for (i = 0; i < 256; i++) {
        if (ht[i]) {
            res++;
            break;
        }
    }
    return res;
}

202. Happy Number

Solution:

bool isHappy(int n) {
    struct Ht *ht = createHT();
    int iht[10] = {0, 1, 4, 9, 16, 25, 36, 49, 64, 81};
    while (n != 1) {
        if (find_or_insert(ht, n)) return false;
        int res = 0;
        while (n) {
            res += iht[n % 10];
            n /= 10;
        }
        n = res;
    }
    return true;
}

Following solution works when the width of int is no more than 64.

#define HTSIZE (9 * 9 * 63)
bool isHappy(int n) {
    int ht[HTSIZE] = {0};
    int iht[10] = {0, 1, 4, 9, 16, 25, 36, 49, 64, 81};
    do {
        int res = 0;
        while (n) {
            res += iht[n % 10];
            n /= 10;
        }
        n = res;
        if (ht[n]) return false;
        ht[n] = 1;
    } while (n != 1);
    return true;
}

36. Valid Sudoku

Solution:

bool isValidSudoku(char** board, int boardRowSize, int boardColSize) {
    int sign[27] = {0};
    int i, j;
    for (i = 0; i < boardRowSize; i++) {
        for (j = 0; j < boardRowSize; j++) {
            if (board[i][j] == '.') continue;
            int flag = 1 << board[i][j];
            if (sign[i] & flag || sign[9 + j] & flag || sign[18 + i / 3 * 3 + j / 3] & flag) return false;
            sign[i] |= flag;
            sign[9 + j] |= flag;
            sign[18 + i / 3 * 3 + j / 3] |= flag;
        }
    }
    return true;
}

1. Two Sum

Solution:

/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
struct He {
    int val;
    int ind;
    struct He *next;
};

struct Ht {
    int size;
    struct He **list;
};

struct Ht *create() {
    struct Ht *ht = malloc(sizeof(struct Ht));
    ht->size = 1000;
    ht->list = calloc(ht->size, sizeof(struct He *));
    return ht;
}

int find(struct Ht *ht, int val) {
    int ind = val > 0 ? val : -val;
    ind = ind % ht->size;
    struct He *p = ht->list[ind];
    while (p) {
        if (p->val == val) return p->ind;
        p = p->next;
    }
    return -1;
}

void insert(struct Ht *ht, int val, int index) {
    int ind = val > 0 ? val : -val;
    ind = ind % ht->size;
    struct He *p = malloc(sizeof(struct He));
    p->val = val;
    p->ind = index;
    p->next = ht->list[ind];
    ht->list[ind] = p;
}

int* twoSum(int* nums, int numsSize, int target) {
    int *res = malloc(2 * sizeof(int));
    struct Ht *ht = create();
    insert(ht, nums[0], 0);
    int i;
    for (i = 1; i < numsSize; i++) {
        int _ind = find(ht, target - nums[i]);
        if (_ind != -1) {
            res[0] = _ind;
            res[1] = i;
            break;
        }
        insert(ht, nums[i], i);
    }
    return res;
}

int singleNumber(int* nums, int numsSize) {
    int i;
    for (i = 1; i < numsSize; i++) nums[0] ^= nums[i];
    return nums[0];
}

170. Two Sum III - Data structure design

Solution:

class TwoSum(object):

    def __init__(self):
        """
        initialize your data structure here
        """
        self.nums = {}


    def add(self, number):
        """
        Add the number to an internal data structure.
        :rtype: nothing
        """
        self.nums[number] = self.nums.get(number, 0) + 1

    def find(self, value):
        """
        Find if there exists any pair of numbers which sum is equal to the value.
        :type value: int
        :rtype: bool
        """
        for num in self.nums:
            theother = value - num
            if num == theother:
                if self.nums[num] > 1:
                    return True
                continue
            if theother in self.nums:
                return True
        return False
