Search a 2D Matrix II

Write an efficient algorithm that searches for a value in an m x n matrix. This matrix has the following properties:

  • Integers in each row are sorted in ascending from left to right.
  • Integers in each column are sorted in ascending from top to bottom.

For example,

  • Consider the following matrix:
[
  [1,   4,  7, 11, 15],
  [2,   5,  8, 12, 19],
  [3,   6,  9, 16, 22],
  [10, 13, 14, 17, 24],
  [18, 21, 23, 26, 30]
]
  • Given target = 5, return true.
  • Given target = 20, return false.

C Solution 1:

bool search(int **matrix, int r1, int c1, int r2, int c2, int target) {
    if (r1 > r2 || c1 > c2) return false;
    if (r2 - r1 < 2 && c2 - c1 < 2) {
        int i, j;
        for (i = r1; i <= r2; i++) {
            for (j = c1; j <= c2; j++) {
                if (matrix[i][j] == target) return true;
            }
        }
        return false;
    }
    int r = r1 + ((r2 - r1) >> 1), c = c1 + ((c2 - c1) >> 1);
    if (target < matrix[r][c]) return search(matrix, r1, c1, r, c, target) || search(matrix, r1, c + 1, r - 1, c2, target) || search(matrix, r + 1, c1, r2, c - 1, target);
    return search(matrix, r, c, r2, c2, target) || search(matrix, r1, c + 1, r - 1, c2, target) || search(matrix, r + 1, c1, r2, c - 1, target);
}
bool searchMatrix(int** matrix, int matrixRowSize, int matrixColSize, int target) {
    if (matrixRowSize < 1 || matrixColSize < 1) return false;
    if (target < matrix[0][0] || target > matrix[matrixRowSize - 1][matrixColSize - 1]) return false;
    if (target == matrix[matrixRowSize - 1][matrixColSize - 1]) return true;
    return search(matrix, 0, 0, matrixRowSize - 1, matrixColSize - 1, target);
}

C Solution 2:

bool searchMatrix(int** matrix, int matrixRowSize, int matrixColSize, int target) {
    int r = 0, c = matrixColSize - 1;
    while (r < matrixRowSize && c > -1) {
        if (matrix[r][c] == target) return true;
        if (target < matrix[r][c]) c--;
        else r++;
    }
    return false;
}
