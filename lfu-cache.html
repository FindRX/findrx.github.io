LFU Cache

Design and implement a data structure for Least Frequently Used (LFU) cache. It should support the following operations: get and put.

  • get(key) - Get the value (will always be positive) of the key if the key exists in the cache, otherwise return -1.
  • put(key, value) - Set or insert the value if the key is not already present. When the cache reaches its capacity, it should invalidate the least frequently used item before inserting a new item. For the purpose of this problem, when there is a tie (i.e., two or more keys that have the same frequency), the least recently used key would be evicted.

Follow up:

  • Could you do both operations in O(1) time complexity?

Example:

LFUCache cache = new LFUCache( 2 /* capacity */ );

cache.put(1, 1);
cache.put(2, 2);
cache.get(1);       // returns 1
cache.put(3, 3);    // evicts key 2
cache.get(2);       // returns -1 (not found)
cache.get(3);       // returns 3.
cache.put(4, 4);    // evicts key 1.
cache.get(1);       // returns -1 (not found)
cache.get(3);       // returns 3
cache.get(4);       // returns 4

Python Solution 1:

class Node(object):
    def __init__(self, val):
        self.val = val
        self.key = None
        self.next = None
        self.pre = None
        self.frequency = None

class LFUCache(object):

    def __init__(self, capacity):
        """
        :type capacity: int
        """
        self.headers = {}
        self.tails = {}
        self.d = {}
        self.capacity = capacity
        self.size = 0
        self.min_frequency = None

    def pick_up(self, node):
        node.next.pre = node.pre
        node.pre.next = node.next
        header = self.headers[node.frequency]
        tail = self.tails[node.frequency]
        if header.next == tail:
            if node.frequency == self.min_frequency:
                self.min_frequency = None

    def put_it_back(self, node):
        if self.min_frequency is None or node.frequency < self.min_frequency:
            self.min_frequency = node.frequency
        if node.frequency not in self.headers:
            self.headers[node.frequency] = Node(0)
            self.tails[node.frequency] = Node(0)
            self.headers[node.frequency].next = self.tails[node.frequency]
            self.tails[node.frequency].pre = self.headers[node.frequency]

        header = self.headers[node.frequency]
        node.next = header.next
        header.next.pre = node
        header.next = node
        node.pre = header

    def get(self, key):
        """
        :type key: int
        :rtype: int
        """
        if self.capacity < 1: return -1
        if key not in self.d: return -1
        node = self.d[key]
        self.pick_up(node)
        node.frequency += 1
        self.put_it_back(node)

        return node.val

    def put(self, key, value):
        """
        :type key: int
        :type value: int
        :rtype: void
        """
        if self.capacity < 1: return -1
        if key in self.d:
            node = self.d[key]
            self.pick_up(node)
            node.val = value
            node.frequency += 1
            self.put_it_back(node)
        elif self.size == self.capacity:
            tail = self.tails[self.min_frequency]
            node = tail.pre
            self.pick_up(node)
            del self.d[node.key]
            node.val = value
            node.key = key
            node.frequency = 1
            self.d[key] = node
            self.put_it_back(node)
        else:
            self.size += 1
            node = Node(value)
            node.key = key
            node.frequency = 1
            self.d[key] = node
            self.put_it_back(node)

# Your LFUCache object will be instantiated and called as such:
# obj = LFUCache(capacity)
# param_1 = obj.get(key)
# obj.put(key,value)

Python Solution 2:

class Node(object):

    def __init__(self, key, val):
        self.key = key
        self.val = val
        self.frequency = 1
        self.pre = None
        self.next = None

class LFUCache(object):

    def __init__(self, capacity):
        """
        :type capacity: int
        """
        self.d = {}
        self.fqs = {}
        self.min_fq = None
        self.size = 0
        self.capacity = capacity

    def _insert(self, node):
        if node.frequency not in self.fqs:
            head = Node(0, 0)
            tail = Node(0, 0)
            head.next, tail.pre = tail, head
            self.fqs[node.frequency] = {'head': head, 'tail': tail}
        head = self.fqs[node.frequency]['head']
        head.next.pre, head.next, node.next, node.pre = node, node, head.next, head

    def get(self, key):
        """
        :type key: int
        :rtype: int
        """
        if self.capacity < 1: return -1
        if key not in self.d: return -1
        node = self.d[key]
        node.next.pre, node.pre.next = node.pre, node.next
        fq = node.frequency
        if fq == self.min_fq and self.fqs[fq]['head'].next == self.fqs[fq]['tail']:
            self.min_fq += 1
        node.frequency += 1
        self._insert(node)
        return node.val

    def put(self, key, value):
        """
        :type key: int
        :type value: int
        :rtype: void
        """
        if self.capacity < 1: return
        if key in self.d:
            node = self.d[key]
            node.val = value
            node.next.pre, node.pre.next = node.pre, node.next
            fq = node.frequency
            if fq == self.min_fq and self.fqs[fq]['head'].next == self.fqs[fq]['tail']:
                self.min_fq += 1
            node.frequency += 1
        elif self.size == self.capacity:
            node = self.fqs[self.min_fq]['tail'].pre
            node.next.pre, node.pre.next = node.pre, node.next
            del self.d[node.key]
            self.d[key] = node
            node.key = key
            node.val = value
            node.frequency = 1
            self.min_fq = 1
        else:
            node = Node(key, value)
            self.min_fq = 1
            self.size += 1
            self.d[key] = node
        self._insert(node)

# obj.put(key,value)
