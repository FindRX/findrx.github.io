Inorder Successor in BST

Given a binary search tree and a node in it, find the in-order successor of that node in the BST.

Note: If the given node has no in-order successor in the tree, return null.

Python Solution 1:

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def inorderSuccessor(self, root, p):
        """
        :type root: TreeNode
        :type p: TreeNode
        :rtype: TreeNode
        """
        pre = [0]
        res = []
        def inorder(root):
            if not root: return
            inorder(root.left)
            if res: return
            if pre[0]:
                res.append(root)
                pre[0] = 0
                return
            if root is p:
                pre[0] = 1
            inorder(root.right)
        inorder(root)
        return res[0] if res else None

Python Solution 3:

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def inorderSuccessor(self, root, p):
        """
        :type root: TreeNode
        :type p: TreeNode
        :rtype: TreeNode
        """

        def inorder(root, found_p):
            if not root: return None, found_p
            left, found_p = inorder(root.left, found_p)
            if left: return left, 0
            if found_p: return root, 0
            if root == p: found_p = 1
            return inorder(root.right, found_p)
        return inorder(root, 0)[0]

Python Solution 4:

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def inorderSuccessor(self, root, p):
        """
        :type root: TreeNode
        :type p: TreeNode
        :rtype: TreeNode
        """
        if not root: return
        if root is p:
            if not root.right: return
            p = root.right
            while p.left:
                p = p.left
            return p
        if p.val > root.val:
            return self.inorderSuccessor(root.right, p)
        node = self.inorderSuccessor(root.left, p)
        return node if node else root
