Add Two Number II

You are given two non-empty linked lists representing two non-negative integers. The most significant digit comes first and each of their nodes contain a single digit. Add the two numbers and return it as a linked list.

  • You may assume the two numbers do not contain any leading zero, except the number 0 itself.
  • What if you cannot modify the input lists? In other words, reversing the lists is not allowed.

Example:

  • Input: 7 -> 2 -> 4 -> 3 + 5 -> 6 -> 4
  • Output: 7 -> 8 -> 0 -> 7

C Solution:

If I can modify the imput.

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */

struct ListNode *reverse(struct ListNode *head) {
    struct ListNode *res = 0;
    while (head) {
        struct ListNode *tmp = head->next;
        head->next = res;
        res = head;
        head = tmp;
    }
    return res;
}
struct ListNode* addTwoNumbers(struct ListNode* l1, struct ListNode* l2) {
    struct ListNode dummy, *p = &dummy;
    l1 = reverse(l1);
    l2 = reverse(l2);
    int carry = 0;
    while (l1 || l2 || carry) {
        p->next = malloc(sizeof(struct ListNode));
        if (l1) {
            carry += l1->val;
            l1 = l1->next;
        }
        if (l2) {
            carry += l2->val;
            l2 = l2->next;
        }
        p->next->val = carry % 10;
        carry /= 10;
        p = p->next;
    }
    p->next = 0;
    return reverse(dummy.next);
}

struct ListNode *reverse(struct ListNode *head) {
    struct ListNode *res = 0;
    while (head) {
        struct ListNode *tmp = head->next;
        head->next = res;
        res = head;
        head = tmp;
    }
    return res;
}
struct ListNode* addTwoNumbers(struct ListNode* l1, struct ListNode* l2) {
    if (!l1) return l2;
    if (!l2) return l1;
    int len1 = 0, len2 = 0;
    struct ListNode *l = l1, *r = l2;
    while (l) {
        len1++;
        l = l->next;
    }
    while (r) {
        len2++;
        r = r->next;
    }

    struct ListNode *dummy = malloc(sizeof(struct ListNode)), *p = dummy;
    dummy->val = 0;

    l = l1;
    r = l2;
    if (len1 > len2) {
        int m = len1 - len2;
        while (m--) {
            p->next = malloc(sizeof(struct ListNode));
            p->next->val = l->val;
            p = p->next;
            l = l->next;
        }
    }
    if (len2 > len1) {
        int m = len2 - len1;
        while (m--) {
            p->next = malloc(sizeof(struct ListNode));
            p->next->val = r->val;
            p = p->next;
            r = r->next;
        }
    }

    while (l) {
        p->next = malloc(sizeof(struct ListNode));
        p->next->val = l->val + r->val;
        p = p->next;
        l = l->next;
        r = r->next;
    }
    p->next = 0;

    dummy = reverse(dummy);
    p = dummy;
    int carry = 0;
    while (p) {
        p->val += carry;
        carry = p->val / 10;
        p->val %= 10;
        p = p->next;
    }
    dummy = reverse(dummy);

    if (dummy->val) return dummy;
    return dummy->next;

}

Python Solution 1:

# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def addTwoNumbers(self, l1, l2):
        """
        :type l1: ListNode
        :type l2: ListNode
        :rtype: ListNode
        """
        def reverse(head):
            head, tail = None, head
            while tail:
                head, tail.next, tail = tail, head, tail.next
            return head

        l1 = reverse(l1)
        l2 = reverse(l2)
        carry = 0
        cur = dummy = ListNode(0)
        while l1 or l2 or carry:
            if l1:
                carry += l1.val
                l1 = l1.next
            if l2:
                carry += l2.val
                l2 = l2.next
            cur.next = cur = ListNode(carry % 10)
            carry //= 10
        return reverse(dummy.next)

Python Solution 2:

# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def addTwoNumbers(self, l1, l2):
        """
        :type l1: ListNode
        :type l2: ListNode
        :rtype: ListNode
        """
        cur = l1
        l1_list = []
        while l1:
            l1_list.append(l1.val)
            l1 = l1.next

        cur = l2
        l2_list = []
        while l2:
            l2_list.append(l2.val)
            l2 = l2.next

        def reverse(head):
            head, tail = None, head
            while tail:
                head, tail.next, tail = tail, head, tail.next
            return head

        cur = dummy = ListNode(0)
        carry = 0
        while l1_list or l2_list or carry:
            carry += l1_list.pop() if l1_list else 0
            carry += l2_list.pop() if l2_list else 0
            cur.next = cur = ListNode(carry % 10)
            carry /= 10
        return reverse(dummy.next)
