Validate Binary Search Tree

Given a binary tree, determine if it is a valid binary search tree (BST).

Assume a BST is defined as follows:

  • The left subtree of a node contains only nodes with keys less than the node's key.
  • The right subtree of a node contains only nodes with keys greater than the node's key.
  • Both the left and right subtrees must also be binary search trees.

Example 1:

    2
   / \
  1   3
  • Binary tree [2,1,3], return true.

Example 2:

    1
   / \
  2   3
  • Binary tree [1,2,3], return false.

C Solution 1:

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
bool helper(struct TreeNode *root, int hasMin, int min, int hasMax, int max) {
    if (!root) return true;
    if (hasMin && min >= root->val) return false;
    if (hasMax && max <= root->val) return false;
    return helper(root->left, hasMin, min, 1, root->val) && helper(root->right, 1, root->val, hasMax, max);
}
bool isValidBST(struct TreeNode* root) {
    return helper(root, 0, 0, 0, 0);
}

C Solution 2:

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
bool helper(struct TreeNode *root, struct TreeNode **pre) {
    if (!root) return true;
    if (!helper(root->left, pre)) return false;
    if (*pre && (*pre)->val >= root->val) return false;
    *pre = root;
    return helper(root->right, pre);
}
bool isValidBST(struct TreeNode* root) {
    struct TreeNode *pre = 0;
    return helper(root, &pre);
}

Python Solution:

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def isValidBST(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        def valid(root, le, lv, re, rv):
            if not root: return True
            if (not le or lv < root.val) and (not re or root.val < rv):
                return valid(root.left, le, lv, True, root.val) and valid(root.right, True, root.val, re, rv)
            return False
        return valid(root, False, None, False, None)
