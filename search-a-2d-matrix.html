Search a 2D Matrix

Write an efficient algorithm that searches for a value in an m x n matrix. This matrix has the following properties:

  • Integers in each row are sorted from left to right.
  • The first integer of each row is greater than the last integer of the previous row.

For example,

  • Consider the following matrix:
[
[1,   3,  5,  7],
[10, 11, 16, 20],
[23, 30, 34, 50]
]
  • Given target = 3, return true.

C Solution 1:

bool searchMatrix(int** matrix, int matrixRowSize, int matrixColSize, int target) {
    if (matrixRowSize < 1 || matrixColSize < 1) return false;
    if (target < matrix[0][0] || target > matrix[matrixRowSize - 1][matrixColSize - 1]) return false;
    int row;
    if (target == matrix[matrixRowSize - 1][0]) return true;
    else if (target > matrix[matrixRowSize - 1][0]) row = matrixRowSize - 1;
    else {
        int l = 0, r = matrixRowSize - 1;
        while (l + 1 < r) {
            int m = l + ((r - l) >> 1);
            if (matrix[m][0] <= target) l = m;
            else r = m;
        }
        row = l;
    }
    if (target == matrix[row][matrixColSize - 1]) return true;
    if (target > matrix[row][matrixColSize - 1]) return false;
    int l = 0, r = matrixColSize - 1;
    while (l + 1 < r) {
        int m = l + ((r - l) >> 1);
        if (matrix[row][m] <= target) l = m;
        else r = m;
    }
    return matrix[row][l] == target;
}

C Solution 2:

bool searchMatrix(int** matrix, int matrixRowSize, int matrixColSize, int target) {
    if (matrixRowSize < 1 || matrixColSize < 1) return false;
    if (target < matrix[0][0] || target > matrix[matrixRowSize - 1][matrixColSize - 1]) return false;
    if (target == matrix[matrixRowSize - 1][matrixColSize - 1]) return true;
    int l = 0, r = matrixRowSize * matrixColSize - 1;
    while (l + 1 < r) {
        int m = l + ((r - l) >> 1);
        if (target < matrix[m / matrixColSize][m % matrixColSize]) r = m;
        else l = m;
    }
    return matrix[l / matrixColSize][l % matrixColSize] == target;
}
