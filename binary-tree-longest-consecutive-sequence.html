Binary Tree Longest Consecutive Sequence

Given a binary tree, find the length of the longest consecutive sequence path.

The path refers to any sequence of nodes from some starting node to any node in the tree along the parent-child connections. The longest consecutive path need to be from parent to child (cannot be the reverse).

For example,

   1
    \
     3
    / \
   2   4
        \
         5

Longest consecutive sequence path is 3-4-5, so return 3.

   2
    \
     3
    / 
   2    
  / 
 1

Longest consecutive sequence path is 2-3,not3-2-1, so return 2.

Python Solution 1:

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def longestConsecutive(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        if not root: return 0
        def recu(root, pre_val, length, res):
            if not root:
                res[0] = max(res[0], length)
                return
            if root.val == pre_val + 1:
                recu(root.left, root.val, length + 1, res)
                recu(root.right, root.val, length + 1, res)
            else:
                res[0] = max(res[0], length)
                recu(root.left, root.val, 1, res)
                recu(root.right, root.val, 1, res)
        res = [0]
        recu(root, root.val - 1, 0, res)
        return res[0]

Python Solution 2:

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def longestConsecutive(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        if not root: return 0
        def recu(root, pre, len, res):
            if root.val == pre + 1:
                len += 1
            else:
                len = 1
            res[0] = max(res[0], len)
            if root.left: recu(root.left, root.val, len, res)
            if root.right: recu(root.right, root.val, len, res)

        res = [0]
        recu(root, root.val - 1, 0, res)
        return res[0]
