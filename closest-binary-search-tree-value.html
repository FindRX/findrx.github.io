Closest Binary Search Tree Value

Given a non-empty binary search tree and a target value, find the value in the BST that is closest to the target.

  • Given target value is a floating point.
  • You are guaranteed to have only one unique value in the BST that is closest to the target.

C Solution 1:

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */

int closestValue(struct TreeNode* root, double target) {
    if (target == root->val) return root->val;
    if (target < root->val) {
        if (!root->left) return root->val;
        int left = closestValue(root->left, target);
        if (root->val - target < fabs(left - target)) return root->val;
        return left;
    }
    if (!root->right) return root->val;
    int right = closestValue(root->right, target);
    if (target - root->val < fabs(right - target)) return root->val;
    return right;
}

C Solution 3:

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
int closestValue(struct TreeNode* root, double target) {
    double diff = fabs(root->val - target);
    int res = root->val;
    while (root) {
        if (root->val == target) return root->val;
        double _diff = fabs(root->val - target);
        if (_diff < diff) {
            res = root->val;
            diff = _diff;
        }
        root = root->val > target ? root->left : root->right;
    }
    return res;
}

C Solution 4:

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
void inorder(struct TreeNode *root, double target, int *left, int *right, int *left_has_value, int *right_has_value) {
    if (target < root->val && root->left) inorder(root->left, target, left, right, left_has_value, right_has_value);
    if (root->val == target) {
        *left = root->val;
        *right = root->val;
        *left_has_value = 1;
        *right_has_value = 1;
    }
    else if (root->val < target) {
        if (!*left_has_value || *left_has_value && *left < root->val) {
            *left = root->val;
            *left_has_value = 1;
        }
    }
    else {
        if (!*right_has_value || *right_has_value && *right > root->val) {
            *right = root->val;
            *right_has_value = 1;
        }
    }
    if (target > root->val && root->right) inorder(root->right, target, left, right, left_has_value, right_has_value);
}
int closestValue(struct TreeNode* root, double target) {
    int left, left_has_value = 0, right, right_has_value = 0;
    inorder(root, target, &left, &right, &left_has_value, &right_has_value);
    if (!left_has_value) return right;
    if (!right_has_value) return left;
    if (target - left > right - target) return right;
    return left;
}

C Solution 5:

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
int closestValue(struct TreeNode* root, double target) {
    if (root->val == target) return root->val;
    if (root->val < target) {
        struct TreeNode *p = root->right;
        if (!p) return root->val;
        while (p->left) p = p->left;
        if (target == p->val) return p->val;
        if (target < p->val) return p->val - target > target - root->val ? root->val : p->val;
        return closestValue(root->right, target);
    }
    struct TreeNode *p = root->left;
    if (!p) return root->val;
    while (p->right) p = p->right;
    if (target == p->val) return p->val;
    if (target > p->val) return target - p->val > root->val - target ? root->val : p->val;
    return closestValue(root->left, target);
}

Python Solution 1:

# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def closestValue(self, root, target):
        """
        :type root: TreeNode
        :type target: float
        :rtype: int
        """
        res = root.val
        if root.val == target:
            return res
        if root.val < target:
            if not root.right: return root.val
            right = self.closestValue(root.right, target)
            return res if abs(res - target) < abs(right - target) else right
        if not root.left: return root.val
        left = self.closestValue(root.left, target)
        return res if abs(res - target) < abs(left - target) else left
