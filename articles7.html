Reverse Linked List

Reverse a singly linked list.

C Solution:

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
struct ListNode* reverseList(struct ListNode* head) {
    struct ListNode *_head = 0;
    while (head) {
        struct ListNode *p = head->next;
        head->next = _head;
        _head = head;
        head = p;
    }
    return _head;
}

Summary:

    int res = 0;
    while (n) res += n = n / 5;
    return res;
