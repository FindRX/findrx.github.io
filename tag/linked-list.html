Reverse Linked List

Reverse a singly linked list.

C Solution 1:

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
struct ListNode* reverseList(struct ListNode* head) {
    struct ListNode *_head = 0;
    while (head) {
        struct ListNode *p = head->next;
        head->next = _head;
        _head = head;
        head = p;
    }
    return _head;
}
